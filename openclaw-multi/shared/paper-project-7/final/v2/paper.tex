% --- Fig 1: Flowchart (after method intro) ---
\begin{figure*}[!t]
\centering
\includegraphics[width=0.95\textwidth]{fig_flowchart.png}
\caption{Overall flowchart of the proposed CAC-CycleGAN-WGP framework. The process consists of four stages: (1) data preprocessing via FFT, (2) cross-domain signal generation using the CycleGAN with CAC and WGP constraints, (3) quality evaluation of synthetic samples via SAE, and (4) fault diagnosis using the SVM classifier.}
\label{fig:flowchart}
\end{figure*}

% --- Fig 2: Architecture ---
\begin{figure*}[!t]
\centering
\includegraphics[width=0.95\textwidth]{fig_architecture.png}
\caption{Detailed architecture of the proposed CAC-CycleGAN-WGP model. The dual-generator structure ($G: X \to Y$, $F: Y \to X$) ensures cycle-consistency, while the Conditional Auxiliary Classifier (CAC) enforces class-specific generation and the Wasserstein distance with Gradient Penalty (WGP) stabilizes training.}
\label{fig:architecture}
\end{figure*}
